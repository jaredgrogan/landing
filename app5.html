Backend with Flask Create a Flask backend to handle the email submission and store the data in an Excel file and JSON files. app.py (Flask backend) python Copy code from flask import Flask, request, jsonify import pandas as pd import json import os app = Flask(__name__) USER_DATA_FOLDER = './user-data' # Ensure the user data folder exists os.makedirs(USER_DATA_FOLDER, exist_ok=True) @app.route('/submit_email', methods=['POST']) def submit_email(): data = request.get_json() email = data.get('email') if not email: return jsonify({'error': 'Email is required'}), 400 # Save email to Excel excel_file = os.path.join(USER_DATA_FOLDER, 'user_emails.xlsx') if os.path.exists(excel_file): df = pd.read_excel(excel_file) else: df = pd.DataFrame(columns=['email']) if email in df['email'].values: return jsonify({'error': 'Email already exists'}), 400 new_row = {'email': email} df = df.append(new_row, ignore_index=True) df.to_excel(excel_file, index=False) # Save user data to JSON user_data = {'email': email} json_file = os.path.join(USER_DATA_FOLDER, f'{email}.json') with open(json_file, 'w') as f: json.dump(user_data, f) return jsonify({'message': 'Email submitted successfully'}), 200 if __name__ == '__main__': app.run(debug=True)
